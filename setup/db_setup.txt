CREATE TABLE `team` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(255) DEFAULT NULL,
  `online_id` int(11) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8

CREATE TABLE `matches` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `league` int(11) DEFAULT NULL,
  `fremad_team` int(11) DEFAULT NULL,
  `home_match` tinyint(1) DEFAULT NULL,
  `home_goals` int(11) DEFAULT NULL,
  `opposing_team_name` varchar(255) DEFAULT NULL,
  `opposing_team_id` int(11) DEFAULT NULL,
  `opposing_team_goals` int(11) DEFAULT NULL,
  `date` date DEFAULT NULL,
  `field` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8

CREATE TABLE `league` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `year` year(4) DEFAULT NULL,
  `team` int(11) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8
